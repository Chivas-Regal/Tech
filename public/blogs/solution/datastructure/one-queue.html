        return g[(x - 1) * m + y - 1];

int main () {
        ios::sync_with_stdio(false);
        cin.tie(nullptr);
 
        cin >> n >> m >> a >> b;
        cin >> g[0] >> x >> y >> z;

                deque<pair<ll, ll> > deq;

                        if (deq.empty()) f[i][j] = h(i, j);
                        else f[i][j] = min(deq.front().second, h(i, j));
                        while (!deq.empty() && deq.back().second >= h(i, j)) deq.pop_back();
                        deq.push_back({j, h(i, j)});

                deque<pair<ll, ll> > deq;

                for (int j = 1; j <= m - b + 1; j ++) {
