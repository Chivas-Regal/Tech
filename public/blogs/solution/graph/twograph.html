    const int N = 510;
int g[N][N], vis[N][N];

int dx[4] = {0, 0, 1, -1};

struct node { int a, b, c; };

        }

    const int N = 4e5 + 10;
int res[N];
struct Edge {
        int nxt, to;
} edge[N];
int head[N], cnt;
int n; 
vector<int> vec[2];

inline void add_Edge ( int from, int to ) {
        edge[ ++ cnt ] = { head[from], to };
        head[from] = cnt;
}
inline void DFS ( int x, int fath, int dep ) {
        vec[dep].push_back(x);
        for ( int i = head[x]; ~i; i = edge[i].nxt ) {
                int to = edge[i].to;
                if ( to == fath ) continue;
                DFS ( to, x, dep^1 );
        }
}

inline void Solve () {
        scanf("%d", &n);

        int stt = 0; while ( (1 << stt) <= n ) stt ++; stt --; // æœ‰å¤šå°‘ä¸ªäºŒè¿›åˆ¶å—
        for ( int i = 0; i <= n * 2 + 5; i ++ ) head[i] = res[i] = -1, cnt = 0;
        vec[0].clear(); vec[1].clear();

        for ( int i = 0; i < n - 1; i ++ ) {
                int a, b; scanf("%d%d", &a, &b);
                add_Edge ( a, b );
                add_Edge ( b, a );
        }
        DFS ( 1, 1, 0 ); // æŸ“è‰²
        for ( ; stt >= 0; stt -- ) {
                int to; // é€‰æ‹©ç‚¹æ•°é‡æœ€å¤šçš„è‰²å—
                if ( max(vec[0].size(), vec[1].size()) == vec[0].size() ) to = 0;

                        res[vec[to].back()] = i;
                        vec[to].pop_back();
                }
        }
        for ( int i = 1; i <= n; i ++ ) printf("%d ", res[i]);
        printf("\n");
}
