        oneSon[ tot ++ ] = { dist, cnt };
        for ( int i = head[x]; ~i; i = edge[i].nxt ) {
                int to = edge[i].to;
                if ( to == fath ) continue;
                get_Dis ( to, x, dist + edge[i].val, cnt + 1, tot );
        }
}

        get_Wc ( x, -1, get_Sz ( x, -1 ), x );

                }
