int n, m;
int c[N], w[N];
struct Edge {
        int nxt, to;

int head[N], cnt;
int vis[N][N];
inline void add_Edge ( int from, int to ) {
        edge[++ cnt] = { head[from], to };
        head[from] = cnt;
